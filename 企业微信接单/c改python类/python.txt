# 定义一个类class_lable_t
class class_lable_t:
        # 初始化函数
        def __init__(self, str, color, height, vidth, ptr):
                self.str = str      # 类别名称
                self.color = color  # 类别的颜色
                self.height = height    # 类别的高度
                self.vidth = vidth      # 类别的宽度
                self.ptr = ptr      # 类别的指针
